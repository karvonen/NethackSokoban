1   package dev.nethacksokoban.ui;
2   
3   import dev.nethacksokoban.game.Game;
4   import java.awt.GridLayout;
5   import java.awt.event.ActionEvent;
6   import java.awt.event.ActionListener;
7   import javax.swing.JButton;
8   import javax.swing.JComboBox;
9   import javax.swing.JLabel;
10  import javax.swing.JPanel;
11  import javax.swing.SwingConstants;
12  
13  /**
14   * Class extends JPanel and creates on it a new grid layout containing the
15   * game's menu.
16   */
17  public class MenuPanel extends JPanel implements ActionListener {
18  
19      private JComboBox<Integer> levels;
20      private GUI gui;
21      private JLabel moves;
22  
23      /**
24       * Menu panel for which includes level selector, move counter and exit
25       * button.
26       *
27       * @param game Game which is controlled by the menu.
28       * @param gui GUI element that needs to be given focus after an event.
29       */
30      public MenuPanel(Game game, GUI gui) {
31          super(new GridLayout(1, 3));
32          this.gui = gui;
33          createComponents();
34      }
35  
36      /**
37       * Creates the menu panel contents.
38       *
39       */
40      private void createComponents() {
41          levels = new JComboBox<>();
42          levels.addActionListener(this);
43          add(levels);
44  
45          moves = new JLabel("Moves: 0", SwingConstants.CENTER);
46          add(moves);
47  
48          JButton exit = new JButton("Exit");
49          exit.addActionListener(this);
50          add(exit);
51      }
52  
53      /**
54       * Adds all the levels to the JComboBox level selector, starting from 1.
55       *
56       * @param levelCount Number of maps the game has loaded.
57       */
58      public void fillLevelSelector(int levelCount) {
59          for (int i = 1; i <= levelCount; i++) {
60              levels.addItem(i);
61          }
62      }
63  
64      /**
65       * Updates the number of legal moves used.
66       *
67       * @param moveCount Number of moves to be displayed.
68       */
69      public void updateMoves(int moveCount) {
70          moves.setText("Moves: " + moveCount);
71      }
72  
73      public JComboBox<Integer> getJComboBox() {
74          return levels;
75      }
76  
77      @Override
78      public void actionPerformed(ActionEvent ae) {
79          if (ae.getActionCommand().equals("Exit")) {
80              System.exit(0);
81          } else if (ae.getActionCommand().equals("comboBoxChanged")) {
82              JComboBox comboBox = (JComboBox) ae.getSource();
83              int selectedNumber = comboBox.getSelectedIndex();
84              gui.startNewMapWithIndex(selectedNumber + 1);
85              gui.setFocusBackToFrame();
86          }
87      }
88  }
